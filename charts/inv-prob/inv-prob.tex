\documentclass{standalone}

\usepackage{pgfplots}

\pgfplotsset{compat=newest}

\begin{document}

\begin{tikzpicture}
  \begin{axis}[
    ylabel = Probability, xlabel = $q$, legend pos = south east,
    legend style = { draw = none, font = \footnotesize },
  ]
    \addplot[densely dotted, thick] gnuplot [raw gnuplot] {
      plot 'prob.txt' using 1:3 every 36
    };
    \addlegendentry{$\Pi(q, n)$}
    \addplot[densely dashed, thick] gnuplot [raw gnuplot] {
      plot 'prob.txt' using 1:6 every 36
    };
    \addlegendentry{$\widetilde{\Pi}(q, n)$}
  \end{axis}
\end{tikzpicture}

\end{document}
